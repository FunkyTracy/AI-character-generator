# --- Application Settings ---
APP_NAME="AI Character Generator"
DEBUG=true

# --- Neo4j Database ---
# Option 1: Local Neo4j Desktop installation
# Default URI for local: bolt://localhost:7687
# 
# Option 2: Neo4j Aura (free cloud tier)
# Get your URI from: https://neo4j.com/cloud/aura/
# Format: neo4j+s://xxxxx.databases.neo4j.io

NEO4J_URI=bolt://localhost:7687
NEO4J_USER=neo4j
NEO4J_PASSWORD=your_neo4j_password_here

# --- OpenAI API ---
# Get your key from: https://platform.openai.com/api-keys
# 
# COST WARNING: API calls cost money!
# gpt-4o-mini is recommended for learning (much cheaper)
# gpt-4o is more capable but more expensive

OPENAI_API_KEY=your_openai_api_key_here
OPENAI_MODEL=gpt-4o-mini

# --- JWT Authentication ---
# Generate a secure key with:
# python -c "import secrets; print(secrets.token_hex(32))"
# 
# IMPORTANT: Use a strong, unique key in production!

JWT_SECRET_KEY=your_secret_key_here_change_in_production
JWT_ALGORITHM=HS256
JWT_EXPIRE_MINUTES=30
